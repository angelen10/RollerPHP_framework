配置文件

Configs/config.php - 入口配置
// 配置默认入口
define('Home','Index'); //此处配置入口控制器目录
define('Cont','index'); //此处配置入口控制器文件
define('Meth','start'); //此处配置入口控制器初始化方法

Configs/db.php - 数据库配置
// 配置数据库
define('DBTYPE','mysql'); //此处配置数据库类型
define('DBNAME','test'); //此处配置数据库名
define('DBHOST','127.0.0.1'); //此处配置IP
define('DBPORT',''); //此处配置端口
define('DBUSER','root'); //此处配置用户名
define('DBPASS','porsche'); //此处配置密码
define('DBCHAR','utf8'); //此处配置字符集
define('DBTbBG',''); //此处配置数据表前缀

级别示意：
// 目录

控制器访问：
在访问时候url具体为：http://host/RollerPHP/index.php?Home=Home文件夹&Cont=控制器文件名(无.php后缀)&Meth=方法名

当无参数时候：http://host/RollerPHP/
默认参数均为Configs/config.php配置文件参数
define('Home','Index'); //此处配置入口控制器目录

入门：Models数据!

数据模型:Models/hello.model.php:
system::load_class('db' , '' , 0);//载入数据库操作类，(类名[，额外路径[，是否new]])
class hello extends db{
  public $PDO_OBJ = null; // PDO对象
  public function __construct() {
    $this->PDO_OBJ = system::load_pdo('', true); // (数据库名称_可以不填则为配置默认[,是否开启memcache])
  }

  public function search() {
    $ResArr = $this->PDO_OBJ->select_Tab('test')->select_Obj('*')->search_command(); // 查询test表内的所有*字段数据
    // var_dump($ResArr);
    return $ResArr;
  }
}

入门：Views视图!

控制器Controller/Index/index.php:
//在页面中访问  .../?Home=Index&Cont=index&Meth=testDrawView
// 渲染html
public function testDrawView() {
  // 参数数组
  $modelArr = array(array('hello' => 'RollerPHP', 'name' => 'myname', 'sex' => 'boy' ),array('hello' => 'RollerPHP2', 'name' => 'myname2', 'sex' => 'boy2' ));

  // 渲染html，drawViews(视图名[,参数数组])
  echo system::drawViews('index',$modelArr)."<br>这是一个html渲染实例";
}
